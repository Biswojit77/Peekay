function doGet() {
  return ContentService.createTextOutput("Web app working");
}

function doPost(e) {
  const type = e.parameter.type;

  if (type === "enroll") {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Registration");

    sheet.appendRow([
      e.parameter.studentName || "",
      e.parameter.grade || "",
      e.parameter.parentName || "",
      e.parameter.phone || "",
      e.parameter.course || "",
      e.parameter.batch || "",
      e.parameter.message || "",
      new Date()
    ]);

    return ContentService.createTextOutput("Enrollment Saved");
  }

  if (type === "feedback") {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Feedback");

    sheet.appendRow([
      e.parameter.name || "",
      e.parameter.role || "",
      e.parameter.rating || "",
      e.parameter.message || "",
      new Date()
    ]);

    return ContentService.createTextOutput("Feedback Saved");
  }

  return ContentService.createTextOutput("Invalid type");
}
